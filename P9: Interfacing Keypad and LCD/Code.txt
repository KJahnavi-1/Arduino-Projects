#include <Keypad.h>
#include <LiquidCrystal.h>

// Defining the no.of rows and columns of the keypad
const byte ROWS = 4; 
const byte COLS = 4; 

//Defining the elements in the rows and columns
char keys[ROWS][COLS] = { 
  {'1','2','3','A'}, 
  {'4','5','6','B'},
  {'7','8','9','C'}, 
  {'*','0','#','D'} 
}; 

//Declaring the Pins to the rows and columns
byte rowPins[ROWS] = {13,12,11,10};
byte colPins[COLS] = {9,8,1,0}; 

// Create a Keypad object using the defined keymap and pin configuration
Keypad customKeypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS,COLS);

//Declaring a variable lcd which consists the pins numbers of rs,en,d4,d5,d6,d7 od LCD
LiquidCrystal lcd(7,6,5,4,3,2);

void setup() { 
  //Defining the rows and columns of the LCD
  lcd.begin(16,2);
  //It prints the defined one on the LCD
  lcd.print("Hi! I'm Jahnavi. Use the keypad.");
  delay(1000); //wait 1 second
  
  for(int i=0;i<16;i++){
    //The Displayed information will start scrolling to left with 200 millisecond in presence
    lcd.scrollDisplayLeft();
    delay(200);
  }
  //The LCD gets cleared 
  lcd.clear(); 
  //The starting point will be set to ROW0 and COLUMN0 in the LCD
  lcd.setCursor(0,0);
}

void loop() 
{ 
  //Reads the key pressed on the keypad
  char customKey = customKeypad.getKey();
  
  //If the customkey pressed then the next lines will be executed
  if(customKey)
  { 
    //It will displays the pressed key in the LCD
    lcd.print(customKey);
  } 
}
